48def differences_by_dim_func(X1, X2, n_order, return_deriv=True): 

49 """ 

50 Compute pairwise differences between two input arrays X1 and X2 for each dimension, 

51 embedding hypercomplex units along each dimension for automatic differentiation. 

52 

53 For each dimension k, this function computes: 

54 diff_k[i, j] = X1[i, k] + e_{k+1} - X2[j, k] 

55 where e_{k+1} is a hypercomplex unit for the (k+1)-th dimension with order 2 * n_order. 

56 

57 Parameters 

58 ---------- 

59 X1 : array_like of shape (n1, d) 

60 First set of input points with n1 samples in d dimensions. 

61 

62 X2 : array_like of shape (n2, d) 

63 Second set of input points with n2 samples in d dimensions. 

64 

65 n_order : int 

66 The base order used to construct hypercomplex units (e_{k+1}) with order 2 * n_order. 

67 

68 index : int, optional 

69 Currently unused. Reserved for future enhancements. 

70 

71 Returns 

72 ------- 

73 differences_by_dim : list of length d 

74 A list where each element is an array of shape (n1, n2), containing the differences 

75 between corresponding dimensions of X1 and X2, augmented with hypercomplex units. 

76 

77 Notes 

78 ----- 

79 - The function leverages hypercomplex arithmetic from the pyOTI library. 

80 - This routine is typically used in the construction of hypercomplex kernels for Gaussian processes 

81 or other applications involving automatic differentiation. 

82 

83 Example 

84 ------- 

85 >>> X1 = [[1.0, 2.0], [3.0, 4.0]] 

86 >>> X2 = [[1.5, 2.5], [3.5, 4.5]] 

87 >>> n_order = 1 

88 >>> diffs = differences_by_dim_func(X1, X2, n_order) 

89 >>> len(diffs) 

90 2 

91 >>> diffs[0].shape 

92 (2, 2) 

93 """ 

94 X1 = oti.array(X1) 

95 X2 = oti.array(X2) 

96 n1, d = X1.shape 

97 

98 n2, d = X2.shape 

99 

100 # Prepare the output: a list of d arrays, each of shape (n, m) 

101 differences_by_dim = [] 

102 

103 # Loop over each dimension k 

104 if n_order == 0: 

105 for k in range(d): 

106 diffs_k = oti.zeros((n1, n2)) 

107 for i in range(n1): 

108 diffs_k[i, :] = ( 

109 X1[i, k] 

110 - (X2[:, k].T) 

111 ) 

112 differences_by_dim.append(diffs_k) 

113 elif not return_deriv: 

114 

115 for k in range(d): 

116 # Create an empty (n, m) array for this dimension 

117 diffs_k = oti.zeros((n1, n2)) 

118 

119 # Nested loops to fill diffs_k 

120 for i in range(n1): 

121 diffs_k[i, :] = ( 

122 X1[i, k] 

123 + oti.e(k + 1, order=n_order) 

124 - (X2[:, k].T) 

125 ) 

126 

127 # Append to our list 

128 differences_by_dim.append(diffs_k) 

129 else: 

130 for k in range(d): 

131 # Create an empty (n, m) array for this dimension 

132 diffs_k = oti.zeros((n1, n2)) 

133 

134 # Nested loops to fill diffs_k 

135 for i in range(n1): 

136 diffs_k[i, :] = ( 

137 X1[i, k] 

138 - (X2[:, k].T) 

139 ) 

140 

141 # Append to our list 

142 differences_by_dim.append(diffs_k+ oti.e(k + 1, order=2*n_order)) 

143 return differences_by_dim

207def deriv_map(nbases, order): 

208 """ 

209 Creates a mapping from (order, index_within_order) to a single 

210 flattened index for all derivative components. 

211 """ 

212 k = 0 

213 map_deriv = [] 

214 for ordi in range(order + 1): 

215 ndir = coti.ndir_order(nbases, ordi) 

216 map_deriv_i = [0] * ndir 

217 for idx in range(ndir): 

218 map_deriv_i[idx] = k 

219 k += 1 

220 map_deriv.append(map_deriv_i) 

221 return map_deriv

240def rbf_kernel( 

241 phi, 

242 phi_exp, 

243 n_order, 

244 n_bases, 

245 der_indices,

247 index=-1, 

248): 

249 """ 

250 Constructs the RBF kernel matrix with derivative entries using an 

251 efficient pre-allocation strategy combined with a single call to 

252 extract all derivative components. 

253 

254 Parameters 

255 ---------- 

256 (Parameters are the same as the original function) 

257 

258 Returns 

259 ------- 

260 K : ndarray 

261 Full RBF kernel matrix with mixed function and derivative entries. 

262 """ 

263 # --- 1. Initial Setup and Efficient Derivative Extraction --- 

264 dh = coti.get_dHelp() 

265 # Create maps to translate derivative specifications to flat indices 

266 der_map = deriv_map(n_bases, 2 * n_order) 

267 der_indices_tr, der_ind_order = transform_der_indices(der_indices, der_map) 

268 

269 # --- 2. Determine Block Sizes and Pre-allocate Matrix --- 

270 n_rows_func, n_cols_func = phi.shape 

271 n_deriv_types = len(der_indices) 

272 n_pts_with_derivs_cols = sum(len(order_indices) for order_indices in index) 

273 n_pts_with_derivs_rows = sum(len(order_indices) for order_indices in index) 

274 total_rows = n_rows_func + n_pts_with_derivs_rows 

275 total_cols = n_cols_func + n_pts_with_derivs_cols 

276 

277 K = np.zeros((total_rows, total_cols)) 

278 

279 # The full content blocks are always the size of the original phi matrix 

280 base_shape = (n_rows_func, n_cols_func) 

281 

282 # --- 3. Fill the Matrix Block by Block --- 

283 

284 # Block (0,0): Function-Function (K_ff) 

285 # The real part is always at index 0 of the flat array 

286 content_full = phi_exp[0].reshape(base_shape) 

287 K[:n_rows_func, :n_cols_func] = content_full * ((-1) ** powers[0]) 

288 

289 # First Block-Row: Function-Derivative (K_fd) 

290 col_offset = n_cols_func 

291 for j in range(n_deriv_types): 

292 flat_idx = der_indices_tr[j] 

293 content_full = phi_exp[flat_idx].reshape(base_shape) 

294 

295 # Get the indices for this derivative order 

296 current_indices = index[j] 

297 

298 # Use fancy indexing to select only the columns at the specified indices 

299 content_sliced = content_full[:, current_indices] 

300 

301 # Number of points with this derivative order 

302 n_pts_this_order = len(current_indices) 

303 

304 K[:n_rows_func, col_offset: col_offset + n_pts_this_order] = content_sliced * ((-1) ** powers[j + 1]) 

305 col_offset += n_pts_this_order 

306 # First Block-Column: Derivative-Function (K_df) 

307 row_offset = n_rows_func 

308 for i in range(n_deriv_types): 

309 flat_idx = der_indices_tr[i] 

310 content_full = phi_exp[flat_idx].reshape(base_shape) 

311 

312 # Get the indices for this derivative order 

313 current_indices = index[i] 

314 

315 # Use fancy indexing to select only the rows at the specified indices 

316 content_sliced = content_full[current_indices, :] 

317 

318 # Number of points with this derivative order 

319 n_pts_this_order = len(current_indices) 

320 

321 K[row_offset: row_offset + n_pts_this_order, :n_cols_func] = content_sliced * ((-1) ** powers[0]) 

322 row_offset += n_pts_this_order 

323 

324 # Inner Blocks: Derivative-Derivative (K_dd) 

325 row_offset = n_rows_func 

326 for i in range(n_deriv_types): 

327 col_offset = n_cols_func 

328 

329 # Get row indices for this derivative order 

330 row_indices = index[i] 

331 n_pts_row = len(row_indices) 

332 

333 for j in range(n_deriv_types): 

334 # Get column indices for this derivative order 

335 col_indices = index[j] 

336 n_pts_col = len(col_indices) 

337 

338 # Multiply the derivative indices to find the correct flat index 

339 imdir1 = der_ind_order[j] 

340 imdir2 = der_ind_order[i] 

341 new_idx, new_ord = dh.mult_dir( 

342 imdir1[0], imdir1[1], imdir2[0], imdir2[1]) 

343 flat_idx = der_map[new_ord][new_idx] 

344 content_full = phi_exp[flat_idx].reshape(base_shape) 

345 

346 # Slice both rows and columns using non-contiguous indices 

347 # Use numpy's ix_ for efficient 2D fancy indexing 

348 content_sliced = content_full[np.ix_(row_indices, col_indices)] 

349 

350 K[row_offset: row_offset + n_pts_row, 

351 col_offset: col_offset + n_pts_col] = content_sliced * ((-1) ** powers[j + 1]) 

352 

353 col_offset += n_pts_col 

354 

355 row_offset += n_pts_row 

356 

357 return K

515def determine_weights(diffs_by_dim, diffs_test, length_scales, kernel_func, sigma_n): 

516 """ 

517 Vectorized version: compute interpolation weights for multiple test points at once. 

518  

519 Parameters 

520 ---------- 

521 diffs_by_dim : list of ndarray 

522 Pairwise differences between training points (by dimension). 

523 diffs_test : list of ndarray 

524 Pairwise differences between test points and training points (by dimension). 

525 Shape: each array is (n_test, n_train) or similar batch dimension. 

526 length_scales : array-like 

527 Kernel hyperparameters. 

528 kernel_func : callable 

529 Kernel function. 

530 sigma_n : float 

531 Noise parameter (if needed). 

532  

533 Returns 

534 ------- 

535 weights_matrix : ndarray of shape (n_test, n_train) 

536 Interpolation weights for each test point. 

537 """ 

538 

539 # Compute K matrix (training covariance) - same for all test points 

540 K = kernel_func(diffs_by_dim, length_scales).real 

541 n_train = K.shape[0] 

542 

543 # Compute r vectors (test-train covariances) for all test points at once 

544 r_all = kernel_func(diffs_test, length_scales).real # shape: (n_test, n_train) 

545 n_test = r_all.shape[0] 

546 

547 # Build augmented system matrix M (same for all test points) 

548 M = np.zeros((n_train + 1, n_train + 1)) 

549 M[:n_train, :n_train] = K 

550 M[:n_train, n_train] = 1 

551 M[n_train, :n_train] = 1 

552 M[n_train, n_train] = 0 

553 

554 # Build augmented RHS for all test points 

555 r_augmented = np.zeros((n_test, n_train + 1)) 

556 r_augmented[:, :n_train] = r_all 

557 r_augmented[:, n_train] = 1 

558 

559 # Solve for all test points at once: M @ solution.T = r_augmented.T 

560 # solution.T has shape (n_train+1, n_test) 

561 solution = np.linalg.solve(M, r_augmented.T) # shape: (n_train+1, n_test) 

562 

563 # Extract weights (exclude Lagrange multiplier) 

564 weights_matrix = solution[:n_train, :].T # shape: (n_test, n_train) 

565 

566 return weights_matrix

578def extract_common_predictions(predictions, indices, common_tuples, include_fvals=True): 

579 """ 

580 Extract rows corresponding to common derivatives. 

581  

582 predictions: shape (num_ders, num_funcs) where row 0 is f_vals 

583 indices: list of derivative indices for this submodel 

584 common_tuples: set of common derivative indices (as tuples) 

585 """ 

586 offset = 1 if include_fvals else 0 

587 extract_rows = [0] if include_fvals else [] 

588 

589 # Build a lookup: tuple -> row index 

590 idx_to_row = {to_tuple(idx): i + offset for i, idx in enumerate(indices)} 

591

593 for common_idx in sorted(common_tuples): 

594 extract_rows.append(idx_to_row[common_idx]) 

595 

596 return predictions[extract_rows]
